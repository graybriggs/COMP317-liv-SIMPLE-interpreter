
<variable> ::= 
<assignable> ::= <variable>
<expression> ::= <numeral> | -<expression> | <expression> + <expression>
		| <expression> - <expression> | <expression> * <expression>
		| <method-body>

<boolean-expression> ::= true | false
		     | <boolean-expression> && <boolean-expression>
		     | <boolean-expression> || <boolean-expression>
		     | !<boolean-expression>
		     | <expression> < <expression>
		     | <expression> == <expresion>

<program> ::= skip | <assignable> := <expression>;
	      | <program><program>
	      | if (<boolean-expression>) { <program> } else { <program> }
	      | while (<boolean-expression>) { <program> }

<method-body> ::= do { <program> return <expression> }


<num> ::= [1-9] | 0
<numeral> ::= <num><numeral>

<char> ::= <ident-char> 
<string> ::= "<char>" | "<char><string>" | "<numeral><string>"

<ident-char> ::= [a-zA-Z] | _
<variable> ::= <ident-char><variable> | <variable><numeral>

<expression> ::= <numeral> | -<expression> | <expression> + <expression>
		| <expression> - <expression> | <expression> * <expression>
		| <method-body> | call <function-name>

<boolean-expression> ::= true | false
		     | <boolean-expression> && <boolean-expression>
		     | <boolean-expression> || <boolean-expression>
		     | !<boolean-expression>
		     | <expression> < <expression>
		     | <expression> == <expresion>

<program> ::= skip | <variable> := <expression>;
	      | <program> <program>
	      | if (<boolean-expression>) { <program> } else { <program> }
	      | while (<boolean-expression>) { <program> }

// extended

<array-decl> ::= <variable> := [<numeral>]
<array-access> ::= <variable>[<expression>]

<fun-arguments> ::= <variable> | <variable>, <function-arguments> | skip
<fun-name> ::= <ident-char><fun-name> | <fun-name><numeral>
<function> ::= function <function-name> (<function-arguments>) { <program> }

<method-body> ::= do { <program> return <expression> }

<throw> ::= throwException(<string>)
<exception> ::= try { <program> } catchException { <program> }
